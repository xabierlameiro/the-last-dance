## Ejemplo de código TypeScript

```ts mark=2,4:6
interface User {
  name: string; // Este campo es obligatorio
  age?: number;
  email: string; // Campo requerido
  role: 'admin' | 'user'; // Union type
  isActive: boolean; // Estado del usuario
}

function createUser(userData: User): User {
  return {
    ...userData,
    isActive: true
  };
}
```
